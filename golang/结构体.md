# 结构体

结构体是一个连续的内存

结构体定义

1. 自定义类型 和 类型别名

```go

//type 后面跟的是类型

type myInt int		//自定义类型
type yourInt = int	//类型别名


var num myInt
num = 100

fmt.Printf("type:%T value:%v\n", num, num) //type:main.myInt value:100


var person struct{
    name string
    age  int
    hobby []string
}

var p person 
p.name = "张三"
p.age  = 18
p.hobby = []string{"篮球","足球"}
fmt.Println(p) // {张三 18 [篮球 足球]}


//匿名结构体
func test(){
    var s struct{
        name string
        age  uint
    }
    
    s.name = "李四"
	s.age = 19
	fmt.Println(s) //{李四 19}

}
// 传入指针类型的结构体
func test1(x *person){
    x.age = 19
}
var p person 
p.name = "张三"
p.age  = 18
p.hobby = []string{"篮球","足球"}
fmt.Println(p) // {张三 18 [篮球 足球]}
test1(p)
fmt.Println(p) // {张三 19 [篮球 足球]}

//创建 指针类型的结构体
var p2 = new(person)
fmt.Printf("%T\n",p2) // *main.person
fmt.Printf("%p",p2)	

//直接初始化赋值
	p := person{
		name:  "赵六",

	}
	fmt.Println(p)
//按照值的顺序初始化赋值

	p := person{
		"赵六",
		10,
		[]string{"呵呵", "哈呵"},
	}
	fmt.Println(p)	//{赵六 10 [呵呵 哈呵]}



// 匿名结构体

type student struct{

  18

```





#### 嵌套结构体

```go
type address struct{
  city string
  province string
}
type person struct{
  name string
  age  int
  address	//这里用匿名结构体
}

func main(){
  p := person{
    name : "张三",
    age  :  18,
    address : address{	//匿名结构体 
      province:"黑龙江",
      city :"哈尔滨",
    },
  }
  fmt.Println(p)				//{张三 18 {哈尔滨 黑龙江}}
  fmt.Println(p.city)		//哈尔滨
}



```
